#include <iostream>
#include <string>
#include <cctype>

using namespace std;

bool is_palindrome(const char word[])
{
    int len = strlen(word);
    for (int i = 0; i < len / 2; i++)
    {
        if (tolower(word[i]) != tolower(word[len - i - 1]))
        {
            return false;
        }
    }
    return true;
}


string reverse(string input)
{
    for (char &c : input)
    {
        if (islower(c))
        {
            c = toupper(c);
        }
        else if (isupper(c))
        {
            c = tolower(c);
        }
    }
    return input;
}

int main()
{
    char word[100];
    cout << "Please enter a word: ";
    cin >> word;

    if (is_palindrome(word))
    {
        cout << "This is a palindrome.\n" << endl;
    }
    else
    {
        cout << "This is not a palindrome.\n" << endl;
    }

    cin.ignore();

    string sentence;
    cout << "Please enter a string below:" << endl;
    getline(cin, sentence);

    string modified = reverse(sentence);
    cout << "Your modified string:" << endl;
    cout << modified << endl;

    return 0;
}
